func ( string,  fiber.Colors) string {
	switch  {
	case fiber.MethodGet:
		return .Cyan
	case fiber.MethodPost:
		return .Green
	case fiber.MethodPut:
		return .Yellow
	case fiber.MethodDelete:
		return .Red
	case fiber.MethodPatch:
		return .White
	case fiber.MethodHead:
		return .Magenta
	case fiber.MethodOptions:
		return .Blue
	default:
		return .Reset
	}
}

func ( int,  fiber.Colors) string {
	switch {
	case  >= fiber.StatusOK &&  < fiber.StatusMultipleChoices:
		return .Green
	case  >= fiber.StatusMultipleChoices &&  < fiber.StatusBadRequest:
		return .Blue
	case  >= fiber.StatusBadRequest &&  < fiber.StatusInternalServerError:
		return .Yellow
	default:
		return .Red
	}
}
