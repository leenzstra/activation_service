// createTagMap function merged the default with the custom tags
func ( *Config) map[string]LogFunc {
	// Set default tags
	 := map[string]LogFunc{
		TagReferer: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Get(fiber.HeaderReferer))
		},
		TagProtocol: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Protocol())
		},
		TagPort: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Port())
		},
		TagIP: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.IP())
		},
		TagIPs: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Get(fiber.HeaderXForwardedFor))
		},
		TagHost: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Hostname())
		},
		TagPath: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Path())
		},
		TagURL: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.OriginalURL())
		},
		TagUA: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Get(fiber.HeaderUserAgent))
		},
		TagBody: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .Write(.Body())
		},
		TagBytesReceived: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return appendInt(, len(.Request().Body()))
		},
		TagBytesSent: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			if .Response().Header.ContentLength() < 0 {
				return appendInt(, 0)
			}
			return appendInt(, len(.Response().Body()))
		},
		TagRoute: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Route().Path)
		},
		TagResBody: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .Write(.Response().Body())
		},
		TagReqHeaders: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			 := make([]string, 0)
			for ,  := range .GetReqHeaders() {
				 = append(, +"="+)
			}
			return .Write([]byte(strings.Join(, "&")))
		},
		TagQueryStringParams: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Request().URI().QueryArgs().String())
		},

		TagBlack: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.App().Config().ColorScheme.Black)
		},
		TagRed: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.App().Config().ColorScheme.Red)
		},
		TagGreen: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.App().Config().ColorScheme.Green)
		},
		TagYellow: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.App().Config().ColorScheme.Yellow)
		},
		TagBlue: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.App().Config().ColorScheme.Blue)
		},
		TagMagenta: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.App().Config().ColorScheme.Magenta)
		},
		TagCyan: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.App().Config().ColorScheme.Cyan)
		},
		TagWhite: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.App().Config().ColorScheme.White)
		},
		TagReset: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.App().Config().ColorScheme.Reset)
		},
		TagError: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			if .ChainErr != nil {
				return .WriteString(.ChainErr.Error())
			}
			return .WriteString("-")
		},
		TagReqHeader: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Get())
		},
		TagHeader: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Get())
		},
		TagRespHeader: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.GetRespHeader())
		},
		TagQuery: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Query())
		},
		TagForm: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.FormValue())
		},
		TagCookie: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Cookies())
		},
		TagLocals: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			switch v := .Locals().(type) {
			case []byte:
				return .Write()
			case string:
				return .WriteString()
			case nil:
				return 0, nil
			default:
				return .WriteString(fmt.Sprintf("%v", ))
			}
		},
		TagStatus: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			if .enableColors {
				 := .App().Config().ColorScheme
				return .WriteString(fmt.Sprintf("%s %3d %s", statusColor(.Response().StatusCode(), ), .Response().StatusCode(), .Reset))
			}
			return appendInt(, .Response().StatusCode())
		},
		TagMethod: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			if .enableColors {
				 := .App().Config().ColorScheme
				return .WriteString(fmt.Sprintf("%s %-7s %s", methodColor(.Method(), ), .Method(), .Reset))
			}
			return .WriteString(.Method())
		},
		TagPid: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Pid)
		},
		TagLatency: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			 := .Stop.Sub(.Start)
			return .WriteString(fmt.Sprintf("%7v", ))
		},
		TagTime: func( Buffer,  *fiber.Ctx,  *Data,  string) (int, error) {
			return .WriteString(.Timestamp.Load().(string))
		},
	}
	// merge with custom tags from user
	if .CustomTags != nil {

		}
