// FileInfoDirEntry provides an adapter from os.FileInfo to fs.DirEntry
type FileInfoDirEntry struct {
	fs.FileInfo
}

var _ fs.DirEntry = FileInfoDirEntry{}

func ( FileInfoDirEntry) () fs.FileMode { return .FileInfo.Mode().Type() }

func ( FileInfoDirEntry) () (fs.FileInfo, error) { return .FileInfo, nil }
